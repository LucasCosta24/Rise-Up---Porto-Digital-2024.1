Table:

Em HTML, a tag <ul> representa uma lista não ordenada. Ela é utilizada para criar uma lista de itens onde a ordem não possui relevância específica. Cada item dentro da lista não ordenada é definido utilizando a tag <li> (item da lista).
No. {{element.posicao}} Nome {{element.nome}} Cor {{element.cor}}
>

Por padrão, os navegadores da web exibirão listas não ordenadas com marcadores (geralmente círculos, quadrados ou discos). É possível estilizar esses marcadores usando CSS se desejar uma aparência diferente.

Code:
HTML
<table class="table-example" cdk-table [dataSource]="dataSource">
<ng-container cdkColumnDef="posicao">
<th cdk-header-cell *cdkHeaderCellDef> No. > </th>
<td cdk-cell *cdkCellDef="let element"> {{element.posicao}</th>
/ng-container

<ng-container cdkColumnDef="nome">
<th cdk-header-cell *cdkHeaderCellDef> Nome > </th>
<td cdk-cell *cdkCellDef="let element"> {{element.nome}</th>
/ng-container

<ng-container cdkColumnDef="cor">
<th cdk-header-cell *cdkHeaderCellDef> Cor > </th>
<td cdk-cell *cdkCellDef="let element"> {{element.cor}</th>
/ng-container

/table
TS
export interface ColorNumbers {
nome: string;
posicao: number;
cor: string;
}
const COLOR_DATA: ColorNumbers[] = [
{posicao: 1, nome: 'Um', cor: 'Vermelho'},
{posicao: 2, nome: 'Dois', cor: 'Amarelo'},
{posicao: 3, nome: 'Três', cor: 'Azul'},
{posicao: 4, nome: 'Quatro', cor: 'Verde'},
{posicao: 5, nome: 'Cinco', cor: 'Roxo'},
{posicao: 6, nome: 'Seis', cor: 'Laranja'},
];
@Component({
selector: 'app-table',
styleUrl: 'table.component.css',
templateUrl: 'table.component.html',
standalone: true,
imports: [CdkTableModule],
})
expor class TableComponent {
displayedColumns: string[] = ['posicao', 'nome', 'cor'];
dataSource = new ExampleDataSource();
}

export class ExampleDataSource extends DataSource<ColorNumbers> {

data = new BehaviorSubject<ColorNumbers[]>(COLOR_DATA);

connect(): Observable<ColorNumbers[]> {
return this.data;
}

disconnect() {}
}
